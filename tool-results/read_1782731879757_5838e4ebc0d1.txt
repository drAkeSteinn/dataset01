    60→/**
    61→ * Build a prompt string for non-ZAI providers based on caption style and params.
    62→ * This is a GENERIC prompt — no hardcoded content themes.
    63→ * The prompt describes an expert art observer analyzing images for LoRA training.
    64→ */
    65→function buildPrompt(params: CaptionParams): string {
    66→  const { triggerWord, captionStyle, captionTemplate, description, imageDescription, vlmAnalysis, colorInfo } = params;
    67→
    68→  let styleInstruction = '';
    69→  if (captionStyle === 'natural') {
    70→    styleInstruction = 'Write a natural language description, 40-100 words. Use complete, flowing sentences that describe what you observe in the image.';
    71→  } else if (captionStyle === 'tags') {
    72→    styleInstruction = 'Write comma-separated tags. List visual elements in order of importance (subject first, then details). Use short descriptive phrases (2-3 words each).';
    73→  } else {
    74→    styleInstruction = captionTemplate || 'Write a descriptive caption for this image.';
    75→  }
    76→
    77→  let prompt = `You are an expert visual analyst observing an image for LoRA training dataset captioning.
    78→
    79→Your task: Carefully observe the image and write an accurate, detailed caption describing what you see.
    80→
    81→CAPTION STYLE: ${styleInstruction}
    82→${triggerWord ? `TRIGGER WORD: Start your caption with "${triggerWord},"` : ''}
    83→
    84→${description ? `DATASET CONTEXT:\n${description}\n\nThis context describes what the dataset is about. Use it to understand the theme and focus of your captions.\n` : ''}${imageDescription ? `IMAGE-SPECIFIC NOTES:\n${imageDescription}\n\nThese are user-provided notes for this specific image. Prioritize this information in your caption.\n` : ''}${vlmAnalysis ? `VISUAL ANALYSIS (from VLM):\n${vlmAnalysis}\n` : ''}${colorInfo ? `COLOR INFO:\n${colorInfo}\n` : ''}
    85→IMPORTANT INSTRUCTIONS:
    86→- Observe the image carefully and describe what you actually see
    87→- Be accurate and specific — do not invent details not present in the image
    88→- Focus on visual elements: subject, appearance, pose, expression, clothing, setting, lighting, style
    89→- Do NOT add commentary, explanations, or meta-text
    90→- Output ONLY the caption text, nothing else
    91→
    92→Write the caption now:`;
    93→
    94→  return prompt;
    95→}

   110→function isOllamaVisionModel(modelName: string): boolean {
   111→  const visionPatterns = [
   112→    'llava', 'bakllava', 'minicpm-v', 'llava-llama3', 'moondream',
   113→    'ovis', 'qwen2-vl', 'cogvlm', 'internvl', 'idefics', 'paligemma',
   114→    'fuyu', 'kosmos', 'cogagent', 'xgen-mm', 'yi-vl', 'deepseek-vl',
   115→    'bunny', 'obsidian', 'immich',
   116→  ];
   117→  const lower = modelName.toLowerCase();
   118→  return visionPatterns.some((p) => lower.includes(p));
   119→}

   121→/**
   122→ * Extract caption text from an LLM response.
   123→ * Handles models that use reasoning_content (like Qwen3.5, DeepSeek-R1, etc.)
   124→ * where the actual output might be in reasoning_content instead of content.
   125→ */
   126→function extractCaptionFromResponse(
   127→  content: string | undefined,
   128→  reasoningContent: string | undefined,
   129→  providerName: string,
   130→  model: string
   131→): string {
   132→  log(`[${providerName}] Extracting caption from response`, {
   133→    model,
   134→    contentLength: content?.length || 0,
   135→    reasoningContentLength: reasoningContent?.length || 0,
   136→    contentPreview: content?.substring(0, 200) || '(empty)',
   137→    reasoningPreview: reasoningContent?.substring(0, 200) || '(empty)',
   138→  });
   139→
   140→  // If content is non-empty, use it
   141→  if (content && content.trim().length > 0) {
   142→    log(`[${providerName}] Using content field (${content.length} chars)`);
   143→    return content.trim();
   144→  }
   145→
   146→  // Content is empty — try reasoning_content as fallback
   147→  // Some models (Qwen3, DeepSeek-R1) put the actual response in reasoning_content
   148→  // when they run out of tokens during reasoning
   149→  if (reasoningContent && reasoningContent.trim().length > 0) {
   150→    log(`[${providerName}] Content empty, extracting from reasoning_content (${reasoningContent.length} chars)`);
   151→
   152→    // Try to find the actual caption in the reasoning content
   153→    // Reasoning models often write their draft caption after phrases like:
   154→    // "caption:", "Draft:", "Final:", or just the trigger word
   155→    const reasoning = reasoningContent.trim();
   156→
   157→    // Look for lines that start with the trigger word or common caption starters
   158→    const lines = reasoning.split('\n');
   159→    const captionCandidates: string[] = [];
   160→
   161→    for (const line of lines) {
   162→      const trimmed = line.trim();
   163→      // Skip empty lines and reasoning markers
   164→      if (!trimmed || trimmed.startsWith('**') || trimmed.startsWith('*')) continue;
   165→      // Look for lines that look like captions (start with common words or trigger patterns)
   166→      if (trimmed.match(/^(n1pl3fk|[a-zA-Z]+,)/) || trimmed.match(/^(caption|final|draft|answer)[:\s]/i)) {
   167→        // Clean up the line
   168→        const cleaned = trimmed.replace(/^(caption|final|draft|answer)[:\s]*/i, '');
   169→        if (cleaned.length > 10) {
   170→          captionCandidates.push(cleaned);
   171→        }
   172→      }
   173→    }
   174→
   175→    if (captionCandidates.length > 0) {
   176→      // Use the last candidate (usually the final draft)
   177→      const caption = captionCandidates[captionCandidates.length - 1];
   178→      log(`[${providerName}] Extracted caption from reasoning: ${caption.substring(0, 100)}...`);
   179→      return caption;
   180→    }
   181→
   182→    // If no clear caption found, try to extract the last meaningful paragraph
   183→    const paragraphs = reasoning.split(/\n\n+/);
   184→    const lastPara = paragraphs[paragraphs.length - 1]?.trim();
   185→    if (lastPara && lastPara.length > 20 && !lastPara.startsWith('**')) {
   186→      log(`[${providerName}] Using last paragraph from reasoning: ${lastPara.substring(0, 100)}...`);
   187→      return lastPara;
   188→    }
   189→
   190→    // Last resort: return the reasoning content itself (truncated)
   191→    log(`[${providerName}] No clear caption found, using raw reasoning (truncated)`);
   192→    return reasoning.substring(0, 500);
   193→  }
   194→
   195→  log(`[${providerName}] WARNING: Both content and reasoning_content are empty!`);
   196→  return '';
   197→}

   199→/**
   200→ * Post-process a generated caption: prepend trigger word if needed, apply template for custom style.
   201→ */
   202→function postProcessCaption(
   203→  rawCaption: string,
   204→  params: CaptionParams
   205→): string {
   206→  let caption = rawCaption.trim();
   207→
   208→  // Remove any leading/trailing quotes
   209→  caption = caption.replace(/^["'`]+|["'`]+$/g, '');
   210→
   211→  // Remove "Caption:" prefix if present
   212→  caption = caption.replace(/^(caption|final caption|answer)[:\s]*/i, '');
   213→
   214→  const settings: CaptionSettings = {
   215→    triggerWord: params.triggerWord,
   216→    captionStyle: params.captionStyle as 'natural' | 'tags' | 'custom',
   217→    captionTemplate: params.captionTemplate,
   218→  };
   219→
   220→  // For custom style, apply template if the LLM didn't already use it
   221→  if (settings.captionStyle === 'custom' && settings.captionTemplate) {
   222→    if (
   223→      caption.includes('{trigger}') ||
   224→      caption.includes('{description}') ||
   225→      caption.includes('{colors}') ||
   226→      caption.includes('{style}')
   227→    ) {
   228→      let colorsStr = '';
   229→      try {
   230→        const colorData = JSON.parse(params.colorInfo);
   231→        colorsStr = colorData.dominant || '';
   232→      } catch {
   233→        colorsStr = params.colorInfo;
   234→      }
   235→
   236→      caption = applyTemplate(caption, {
   237→        trigger: settings.triggerWord,
   238→        description: params.vlmAnalysis,
   239→        colors: colorsStr,
   240→        style: '',
   241→      });
   242→    }
   243→  }
   244→
   245→  // Prepend trigger word if style is natural or tags and trigger word exists
   246→  if (
   247→    settings.triggerWord &&
   248→    settings.captionStyle !== 'custom' &&
   249→    !caption.toLowerCase().startsWith(settings.triggerWord.toLowerCase())
   250→  ) {
   251→    caption = `${settings.triggerWord}, ${caption}`;
   252→  }
   253→
   254→  return caption;
   255→}

   267→  async generateCaption(params: CaptionParams): Promise<string> {
   268→    const {
   269→      imagePath,
   270→      vlmAnalysis: existingAnalysis,
   271→      colorInfo,
   272→      triggerWord,
   273→      captionStyle,
   274→      captionTemplate,
   275→      description,
   276→      imageDescription,
   277→    } = params;
   278→
   279→    log('[ZAI] Starting caption generation', {
   280→      imagePath,
   281→      hasVlmAnalysis: !!existingAnalysis,
   282→      hasColorInfo: !!colorInfo,
   283→      hasImageDescription: !!imageDescription,
   284→      hasDescription: !!description,
   285→      triggerWord,
   286→      captionStyle,
   287→    });
   288→
   289→    let vlmAnalysis = existingAnalysis;
   290→
   291→    // Step 1: If no VLM analysis exists, run VLM analysis first
   292→    if (!vlmAnalysis) {
   293→      if (!fs.existsSync(imagePath)) {
   294→        throw new Error(`Image file not found: ${imagePath}`);
   295→      }
   296→
   297→      let success = false;
   298→      let cropRatio = 0.20;
   299→
   300→      while (!success && cropRatio > 0.05) {
   301→        try {
   302→          const headCrop = await cropHead(imagePath, cropRatio);
   303→          const base64Image = headCrop.toString('base64');
   304→          const dataUrl = `data:image/jpeg;base64,${base64Image}`;
   305→
   306→          const zai = await getZAI();
   307→          const vlmResponse = await zai.chat.completions.createVision({
   308→            model: 'gpt-4o',
   309→            messages: [
   310→              {
   311→                role: 'system',
   312→                content:
   313→                  'You are an expert image analyst for LoRA training datasets. Describe the subject in detail including: appearance, features, clothing, pose, expression, and any distinctive characteristics. Be thorough and specific.',
   314→              },
   315→              {
   316→                role: 'user',
   317→                content: [
   318→                  {
   319→                    type: 'text',
   320→                    text: 'Describe this image in detail for a LoRA training dataset. Focus on the subject\'s visual characteristics.',
   321→                  },
   322→                  {
   323→                    type: 'image_url',
   324→                    image_url: { url: dataUrl },
   325→                  },
   326→                ],
   327→              },
   328→            ],
   329→          });
   330→
   331→          if (vlmResponse?.choices?.[0]?.message?.content) {
   332→            vlmAnalysis = vlmResponse.choices[0].message.content;
   333→            success = true;
   334→            log('[ZAI] VLM analysis completed', { length: vlmAnalysis.length });
   335→          } else if (
   336→            vlmResponse?.content_filter?.some(
   337→              (f: { level: number }) => f.level === 1301
   338→            )
   339→          ) {
   340→            if (cropRatio > 0.08) {
   341→              cropRatio = 0.08;
   342→              continue;
   343→            } else {
   344→              throw new Error('Content filter triggered even with minimal crop');
   345→            }
   346→          } else {
   347→            throw new Error('Empty VLM response');
   348→          }
   349→        } catch (err: unknown) {
   350→          const errorObj = err as { status?: number; message?: string };
   351→          if (errorObj.status === 429) {
   352→            await new Promise((resolve) => setTimeout(resolve, 25000));
   353→            continue;
   354→          }
   355→          if (
   356→            errorObj.message?.includes('1301') ||
   357→            errorObj.message?.includes('content_filter')
   358→          ) {
   359→            if (cropRatio > 0.08) {
   360→              cropRatio = 0.08;
   361→              continue;
   362→            }
   363→          }
   364→          throw err;
   365→        }
   366→      }
   367→
   368→      if (!success) {
   369→        throw new Error('VLM analysis failed after retries');
   370→      }
   371→    }
   372→
   373→    // Step 2: Use LLM to generate caption from analysis
   374→    const settings: CaptionSettings = {
   375→      triggerWord,
   376→      captionStyle: captionStyle as 'natural' | 'tags' | 'custom',
   377→      captionTemplate,
   378→      description,
   379→    };
   380→
   381→    const systemPrompt = buildSystemPrompt(settings);
   382→    const userPrompt = buildUserPrompt(vlmAnalysis, colorInfo, imageDescription);
   383→
   384→    log('[ZAI] Sending caption generation request', {
   385→      systemPromptLength: systemPrompt.length,
   386→      userPromptLength: userPrompt.length,
   387→      hasDescription: !!description,
   388→      hasImageDescription: !!imageDescription,
   389→    });
   390→
   391→    const zai = await getZAI();
   392→    const response = await zai.chat.completions.create({
   393→      model: 'gpt-4o',
   394→      messages: [
   395→        { role: 'system', content: systemPrompt },
   396→        { role: 'user', content: userPrompt },
   397→      ],
   398→    });
   399→
   400→    const rawCaption = response?.choices?.[0]?.message?.content?.trim() || '';
   401→    log('[ZAI] Caption generated', { rawLength: rawCaption.length, preview: rawCaption.substring(0, 100) });
   402→
   403→    return postProcessCaption(rawCaption, params);
   404→  },
   405→};

   459→      if (!response.ok) {

   463→
   464→      const data = (await response.json()) as { response?: string };
   465→      rawCaption = data.response || '';
   466→      log('[Ollama] Response received', { rawLength: rawCaption.length, preview: rawCaption.substring(0, 100) });
   467→    } else {
   468→      const body = {
   469→        model,
   470→        messages: [
   471→          {
   472→            role: 'system',
   473→            content: 'You are an expert visual analyst creating captions for LoRA training datasets. Observe carefully and describe accurately.',
   474→          },
   475→          { role: 'user', content: prompt },
   476→        ],
   477→        stream: false,
   478→        options: {
   479→          temperature: 0.7,
   480→          num_predict: 500,
   481→        },
   482→      };
   483→
   484→      log('[Ollama] Sending text request', { model, promptPreview: prompt.substring(0, 200) });
   485→
   486→      const response = await fetch(`${baseUrl}/api/chat`, {
   487→        method: 'POST',
   488→        headers: { 'Content-Type': 'application/json' },
   489→        body: JSON.stringify(body),
   490→        signal: AbortSignal.timeout(120000),
   491→      });
   492→
   493→      if (!response.ok) {
   494→        const errorText = await response.text().catch(() => 'Unknown error');
   495→        throw new Error(`Ollama API error (${response.status}): ${errorText}`);
   496→      }
   497→
   498→      const data = (await response.json()) as {
   499→        message?: { content?: string };
   500→      };
   501→      rawCaption = data.message?.content || '';
   502→      log('[Ollama] Response received', { rawLength: rawCaption.length, preview: rawCaption.substring(0, 100) });
   503→    }
   504→
   505→    return postProcessCaption(rawCaption, params);
   506→  },

   587→      signal: AbortSignal.timeout(120000),
   588→    });

   598→        message?: {

   600→          reasoning_content?: string;
   601→        };
   602→        finish_reason?: string;
   603→      }>;
   604→      usage?: {
   605→        prompt_tokens?: number;
   606→        completion_tokens?: number;
   607→        total_tokens?: number;
   608→        completion_tokens_details?: { reasoning_tokens?: number };
   609→      };
   610→    };
   611→
   612→    log('[LM Studio] Response received', {
   613→      finishReason: data.choices?.[0]?.finish_reason,
   614→      usage: data.usage,
   615→      contentLength: data.choices?.[0]?.message?.content?.length || 0,
   616→      reasoningContentLength: data.choices?.[0]?.message?.reasoning_content?.length || 0,
   617→    });
   618→
   619→    // Extract caption — handle reasoning models that put output in reasoning_content
   620→    const content = data.choices?.[0]?.message?.content;
   621→    const reasoningContent = data.choices?.[0]?.message?.reasoning_content;
   622→
   623→    const rawCaption = extractCaptionFromResponse(content, reasoningContent, 'LM Studio', model || 'default');
   624→
   625→    log('[LM Studio] Final caption', {
   626→      rawLength: rawCaption.length,
   627→      preview: rawCaption.substring(0, 200),
   628→    });
   629→
   630→    return postProcessCaption(rawCaption, params);
   631→  },
   632→};

   634→// ─── TextGen WebUI Provider ──────────────────────────────────────────────────
   635→
   636→const textgenProvider: LLMProvider = {
   637→  name: 'Text Generation WebUI',
   638→  id: 'textgen',
   639→  needsEndpoint: true,
   640→  needsModel: true,
   641→  supportsVision: true,
   642→  defaultEndpoint: 'http://localhost:5000',
   643→
   644→  async generateCaption(params: CaptionParams): Promise<string> {
   645→    const { imagePath, model, endpoint } = params;
   646→    const baseUrl = endpoint || textgenProvider.defaultEndpoint;
   647→
   648→    if (!fs.existsSync(imagePath)) {
   649→      throw new Error(`Image file not found: ${imagePath}`);
   650→    }
   651→
   652→    const base64Image = await imageToBase64(imagePath);
   653→    const prompt = buildPrompt(params);
   654→
   655→    log('[TextGen] Starting caption generation', {
   656→      model, endpoint: baseUrl,
   657→      hasDescription: !!params.description,
   658→      hasImageDescription: !!params.imageDescription,
   659→      promptLength: prompt.length,
   660→    });
   661→
   662→    const userContent: Array<
   663→      | { type: 'text'; text: string }
   664→      | { type: 'image_url'; image_url: { url: string } }
   665→    > = [
   666→      { type: 'text', text: prompt },
   667→      {
   668→        type: 'image_url',
   669→        image_url: { url: `data:image/jpeg;base64,${base64Image}` },
   670→      },
   671→    ];
   672→
   673→    const body: Record<string, unknown> = {
   674→      model: model || 'default',
   675→      messages: [
   676→        {
   677→          role: 'system',
   678→          content: 'You are an expert visual analyst creating captions for LoRA training datasets. Observe the image carefully and describe what you see accurately. Output ONLY the caption text.',
   679→        },
   680→        { role: 'user', content: userContent },
   681→      ],
   682→      max_tokens: 2000,
   683→      temperature: 0.7,
   684→    };
   685→
   686→    log('[TextGen] Sending request', { url: `${baseUrl}/api/v1/chat/completions`, promptPreview: prompt.substring(0, 200) });
   687→
   688→    const response = await fetch(`${baseUrl}/api/v1/chat/completions`, {
   689→      method: 'POST',
   690→      headers: { 'Content-Type': 'application/json' },
   691→      body: JSON.stringify(body),
   692→      signal: AbortSignal.timeout(120000),
   693→    });
   694→
   695→    if (!response.ok) {
   696→      const errorText = await response.text().catch(() => 'Unknown error');
   697→      log('[TextGen] API error', { status: response.status, errorText });
   698→      throw new Error(`TextGen WebUI API error (${response.status}): ${errorText}`);
   699→    }
   700→
   701→    const data = (await response.json()) as {
   702→      choices?: Array<{
   703→        message?: {
   704→          content?: string;
   705→          reasoning_content?: string;
   706→        };
   707→        finish_reason?: string;
   708→      }>;
   709→    };
   710→
   711→    const content = data.choices?.[0]?.message?.content;
   712→    const reasoningContent = data.choices?.[0]?.message?.reasoning_content;
   713→
   714→    const rawCaption = extractCaptionFromResponse(content, reasoningContent, 'TextGen', model || 'default');
   715→
   716→    log('[TextGen] Final caption', { rawLength: rawCaption.length, preview: rawCaption.substring(0, 200) });
   717→
   718→    return postProcessCaption(rawCaption, params);
   719→  },
   720→};

   731→export function getProvider(id: string): LLMProvider {
   732→  const provider = PROVIDERS[id];
   733→  if (!provider) {
   734→    throw new Error(`Unknown LLM provider: ${id}. Available: ${Object.keys(PROVIDERS).join(', ')}`);
   735→  }
   736→  return provider;
   737→}
