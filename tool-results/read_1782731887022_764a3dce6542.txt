   126тЖТ   126тЖТfunction extractCaptionFromResponse(

   132тЖТ   132тЖТ  log(`[${providerName}] Extracting caption from response`, {
   133тЖТ   133тЖТ    model,
   134тЖТ   134тЖТ    contentLength: content?.length || 0,
   135тЖТ   135тЖТ    reasoningContentLength: reasoningContent?.length || 0,
   136тЖТ   136тЖТ    contentPreview: content?.substring(0, 200) || '(empty)',
   137тЖТ   137тЖТ    reasoningPreview: reasoningContent?.substring(0, 200) || '(empty)',
   138тЖТ   138тЖТ  });
   139тЖТ   139тЖТ

   141тЖТ   141тЖТ  if (content && content.trim().length > 0) {

   143тЖТ   143тЖТ    return content.trim();
   144тЖТ   144тЖТ  }
   145тЖТ   145тЖТ
   146тЖТ   146тЖТ  // Content is empty тАФ try reasoning_content as fallback
   147тЖТ   147тЖТ  // Some models (Qwen3, DeepSeek-R1) put the actual response in reasoning_content
   148тЖТ   148тЖТ  // when they run out of tokens during reasoning
   149тЖТ   149тЖТ  if (reasoningContent && reasoningContent.trim().length > 0) {
   150тЖТ   150тЖТ    log(`[${providerName}] Content empty, extracting from reasoning_content (${reasoningContent.length} chars)`);
   151тЖТ   151тЖТ
   152тЖТ   152тЖТ    // Try to find the actual caption in the reasoning content
   153тЖТ   153тЖТ    // Reasoning models often write their draft caption after phrases like:
   154тЖТ   154тЖТ    // "caption:", "Draft:", "Final:", or just the trigger word
   155тЖТ   155тЖТ    const reasoning = reasoningContent.trim();
   156тЖТ   156тЖТ
   157тЖТ   157тЖТ    // Look for lines that start with the trigger word or common caption starters
   158тЖТ   158тЖТ    const lines = reasoning.split('\n');
   159тЖТ   159тЖТ    const captionCandidates: string[] = [];
   160тЖТ   160тЖТ
   161тЖТ   161тЖТ    for (const line of lines) {
   162тЖТ   162тЖТ      const trimmed = line.trim();
   163тЖТ   163тЖТ      // Skip empty lines and reasoning markers
   164тЖТ   164тЖТ      if (!trimmed || trimmed.startsWith('**') || trimmed.startsWith('*')) continue;
   165тЖТ   165тЖТ      // Look for lines that look like captions (start with common words or trigger patterns)
   166тЖТ   166тЖТ      if (trimmed.match(/^(n1pl3fk|[a-zA-Z]+,)/) || trimmed.match(/^(caption|final|draft|answer)[:\s]/i)) {
   167тЖТ   167тЖТ        // Clean up the line
   168тЖТ   168тЖТ        const cleaned = trimmed.replace(/^(caption|final|draft|answer)[:\s]*/i, '');
   169тЖТ   169тЖТ        if (cleaned.length > 10) {
   170тЖТ   170тЖТ          captionCandidates.push(cleaned);
   171тЖТ   171тЖТ        }
   172тЖТ   172тЖТ      }
   173тЖТ   173тЖТ    }
   174тЖТ   174тЖТ
   175тЖТ   175тЖТ    if (captionCandidates.length > 0) {
   176тЖТ   176тЖТ      // Use the last candidate (usually the final draft)
   177тЖТ   177тЖТ      const caption = captionCandidates[captionCandidates.length - 1];
   178тЖТ   178тЖТ      log(`[${providerName}] Extracted caption from reasoning: ${caption.substring(0, 100)}...`);
   179тЖТ   179тЖТ      return caption;
   180тЖТ   180тЖТ    }
   181тЖТ   181тЖТ
   182тЖТ   182тЖТ    // If no clear caption found, try to extract the last meaningful paragraph
   183тЖТ   183тЖТ    const paragraphs = reasoning.split(/\n\n+/);
   184тЖТ   184тЖТ    const lastPara = paragraphs[paragraphs.length - 1]?.trim();
   185тЖТ   185тЖТ    if (lastPara && lastPara.length > 20 && !lastPara.startsWith('**')) {
   186тЖТ   186тЖТ      log(`[${providerName}] Using last paragraph from reasoning: ${lastPara.substring(0, 100)}...`);
   187тЖТ   187тЖТ      return lastPara;
   188тЖТ   188тЖТ    }
   189тЖТ   189тЖТ
   190тЖТ   190тЖТ    // Last resort: return the reasoning content itself (truncated)
   191тЖТ   191тЖТ    log(`[${providerName}] No clear caption found, using raw reasoning (truncated)`);
   192тЖТ   192тЖТ    return reasoning.substring(0, 500);
   193тЖТ   193тЖТ  }
   194тЖТ   194тЖТ
   195тЖТ   195тЖТ  log(`[${providerName}] WARNING: Both content and reasoning_content are empty!`);
   196тЖТ   196тЖТ  return '';
   197тЖТ   197тЖТ}
